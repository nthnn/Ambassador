18package web.ambassador.view;
19
20import web.ambassador.core.AssetResponse;
21
22/**
23 * Represents the content of an asset that can be rendered as part of a web response.
24 *
25 * The {@code AssetContent} class implements the {@code ViewContent} interface and is used to
26 * represent static assets such as images, JavaScript files, CSS files, or other types of media.
27 * The class holds an {@code AssetResponse} object, which encapsulates the asset's content and MIME type.
28 * It provides methods to render the asset content as a byte array and retrieve the appropriate MIME type
29 * for the asset.
30 *
31 * The class employs a simple caching mechanism to avoid redundant content retrieval. The asset content
32 * is fetched only once and stored in memory for subsequent access, improving performance for repeated requests
33 * for the same asset.
34 *
35 * @see web.ambassador.view.ViewContent
36 * @see web.ambassador.core.AssetResponse
37 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
38 * @since 1.0
39 * @version 1.0
40 */
41public class AssetContent implements ViewContent {
42 /** Cached content to avoid redundant fetching. */
43 private byte[] cache = null;
44
45 /** The AssetResponse that holds the asset's content and MIME type. */
46 private final AssetResponse assetResponse;
47
48 /**
49 * Constructs a new {@code AssetContent} instance.
50 *
51 * This constructor initializes the {@code AssetContent} with the given {@code AssetResponse}
52 * which contains the asset's content and MIME type.
53 *
54 * @param assetResponse The {@code AssetResponse} containing the asset's content and MIME type.
55 */
56 public AssetContent(AssetResponse assetResponse) {
57 this.assetResponse = assetResponse;
58 }
59
60 /**
61 * Renders the content of the asset.
62 *
63 * If the content has already been cached, it returns the cached byte array. Otherwise, it fetches
64 * the content from the {@code AssetResponse}, stores it in the cache, and then returns it.
65 *
66 * @return A byte array representing the rendered asset content, ready to be sent as part of the HTTP response body.
67 */
68 @Override
69 public byte[] render() {
70 if(this.cache == null)
71 this.cache = this.assetResponse.getContent();
72
73 return this.cache;
74 }
75
76 /**
77 * Returns the MIME type of the asset content.
78 *
79 * The MIME type indicates the type of content represented by this asset. For example, an image may
80 * have the MIME type {@code "image/png"} or {@code "image/jpeg"}, while a CSS file might have
81 * the MIME type {@code "text/css"}. This is useful for proper content handling by the client (e.g., a browser).
82 *
83 * @return A string representing the MIME type of the asset (e.g., "image/png", "text/css").
84 */
85 @Override
86 public String mimeType() {
87 return this.assetResponse.getMimeType();
88 }
89
90 /**
91 * Returns a string representation of the asset content.
92 *
93 * This method converts the asset content to a string using the {@code render} method. The result
94 * may not be meaningful for binary assets (such as images), but is implemented for debugging or logging purposes.
95 *
96 * @return A string representation of the asset content, or a human-readable form if possible.
97 */
98 @Override
99 public String toString() {
100 return new String(this.render());
101 }
102}
