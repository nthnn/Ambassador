25/**
26 * The AssetHandler class is responsible for handling the retrieval of static assets
27 * (e.g., images, CSS files, JavaScript files) from the classpath and returning them
28 * along with the appropriate MIME type for proper content handling in web responses.
29 *
30 * This class contains the logic for:
31 * <ul>
32 * <li>Fetching the asset's content from the classpath using the provided path.</li>
33 * <li>Determining the asset's MIME type based on its file extension.</li>
34 * <li>Returning an AssetResponse containing the asset content and MIME type.</li>
35 * </ul>
36 *
37 * The class also includes a static map of known file extensions to MIME types, which helps in setting the correct content type for various asset files.
38 *
39 * <p><b>Usage:</b></p>
40 * ```java
41 * AssetHandler assetHandler = new AssetHandler();
42 * AssetResponse response = assetHandler.getAsset("/assets/image.png");
43 *
44 * if(response.wasFound()) {
45 * // Process the content and MIME type
46 * }
47 * ```
48 *
49 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
50 * @since 1.0
51 * @version 1.0
52 * @see web.ambassador.core.AssetResponse
53 */
54public class AssetHandler {
55 private static final Map<String, String> mimeTypes = new HashMap<>();
56
57 /**
58 * Retrieves an asset from the classpath based on the provided path and returns
59 * the asset content along with its MIME type.
60 * <p>
61 * The method strips the `/assets/` prefix from the path, sanitizes the path
62 * to avoid directory traversal attacks, and attempts to locate the asset
63 * from the classpath.
64 * </p>
65 *
66 * @param path The path to the asset (must start with "/assets/").
67 * @return An {@code AssetResponse} containing the asset content, MIME type,
68 * and whether the asset was found and successfully loaded.
69 */
70 public AssetResponse getAsset(String path) {
71 path = path.replaceFirst("^/assets/", "");
72 path = "assets/" + path.replace("..", "");
73
74 try {
75 InputStream inputStream = getClass().getClassLoader().getResourceAsStream(path);
76 if(inputStream == null)
77 return new AssetResponse(null, null, false);
78
79 byte[] content = inputStream.readAllBytes();
80 inputStream.close();
81
82 String extension = "";
83 int i = path.lastIndexOf('.');
84 if(i > 0)
85 extension = path.substring(i + 1).toLowerCase();
86
87 String mimeType = AssetHandler.mimeTypes.getOrDefault(
88 extension,
89 "application/octet-stream"
90 );
91 return new AssetResponse(
92 content,
93 mimeType,
94 true
95 );
96 }
97 catch(IOException e) {
98 return new AssetResponse(null, null,false);
99 }
100 }
