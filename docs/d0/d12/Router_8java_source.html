33/**
34 * The {@code Router} class handles the registration of controller classes and
35 * maps incoming HTTP requests to the appropriate route handlers. It acts as
36 * the entry point for processing web requests in the application.
37 * <br/>
38 * The {@code Router} supports dynamic registration of routes through controller
39 * annotations, enabling flexible and scalable request handling.
40 *
41 * <p><b>Key Responsibilities:</b></p>
42 * <ul>
43 * <li>Register controller classes and associate routes dynamically.</li>
44 * <li>Route incoming requests to the correct handler based on path and method.</li>
45 * <li>Handle static asset requests (e.g., CSS, JS, images) through an {@code AssetHandler}.</li>
46 * <li>Return appropriate HTTP responses (404, 405) for unmatched or invalid requests.</li>
47 * </ul>
48 *
49 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
50 * @version 1.0
51 * @since 1.0
52 */
53public class Router {
54 private final List<Route> routes;
55 private final AssetHandler assetHandler;
56
57 /**
58 * Constructs a new {@code Router} instance with an empty list of routes
59 * and an asset handler for serving static files.
60 */
61 public Router() {
62 this.routes = new ArrayList<>();
63 this.assetHandler = new AssetHandler();
64 }
65
66 /**
67 * Registers a controller class and extracts route mappings from its
68 * annotated methods.
69 *
70 * This method scans the provided controller class for {@code @Method} annotations
71 * and registers the corresponding routes dynamically. The base path is extracted
72 * from the {@code @Controller} annotation.
73 *
74 * @param controllerClass The controller class to register.
75 */
76 public void registerController(Class<?> controllerClass) {
77 Controller controllerAnnotation = controllerClass.getAnnotation(Controller.class);
78 if(controllerAnnotation == null)
79 return;
80
81 String basePath = controllerAnnotation.path();
82 Route route = new Route(basePath);
83
84 for(Method method : controllerClass.getMethods()) {
85 web.ambassador.annotations.Method methodAnnotation =
86 method.getAnnotation(web.ambassador.annotations.Method.class);
87
88 if(methodAnnotation != null)
89 route.addHandler(methodAnnotation.value(),
90 new RouteHandler(controllerClass, method));
91 }
92
93 this.routes.add(route);
94 }
95
96 /**
97 * Handles incoming HTTP requests by matching the request path to a registered route.
98 *
99 * If the request path starts with {@code /assets/}, the request is forwarded
100 * to the {@code AssetHandler} to retrieve static assets.
101 * <br/>
102 * For other paths, the router attempts to find a matching route and invokes
103 * the associated route handler. If no match is found, a 404 response is returned.
104 * If the method is not allowed, a 405 response is generated.
105 *
106 * @param request The incoming HTTP request.
107 * @param response The HTTP response to populate.
108 * @param dbManager The database manager to provide access to data.
109 * @return A {@code ViewContent} representing the response body (HTML, JSON, etc.).
110 */
111 public ViewContent handleRequest(Request request, Response response, DatabaseManager dbManager) {
112 String path = request.getPath();
113
114 if(path.startsWith("/assets/")) {
115 AssetResponse assetResponse = assetHandler.getAsset(path);
116 if(assetResponse.wasFound()) {
117 response.getHeaders().put(
118 "Content-Type",
119 assetResponse.getMimeType()
120 );
121 return new AssetContent(assetResponse);
122 }
123
124 response.setStatusCode(404);
125 return Dom.h1("404 Asset Not Found");
126 }
127
128 for(Route route : routes)
129 if(route.matches(path)) {
130 Map<String, String> pathParams = route.extractParams(path);
131 request.getPathParams().putAll(pathParams);
132
133 RouteHandler handler = route.getHandler(request.getMethod());
134 if(handler != null)
135 return handler.handle(request, response, dbManager);
136
137 response.setStatusCode(405);
138 return Dom.h1("405 Method Not Allowed");
139 }
140
141 response.setStatusCode(404);
142 return Dom.h1("404 Not Found");
143 }
144}
