20import java.sql.ResultSet;
21import java.sql.SQLException;

23/**
24 * A functional interface for mapping rows of a {@code ResultSet} to an object of type {@code T}.
25 *
26 * This interface provides a way to map each row of the {@code ResultSet} to a specific object. It is commonly used
27 * in JDBC-based data access layers where each row of a query result is mapped to a Java object. The implementation
28 * of this interface can be customized to map the rows according to the specific needs of the application.
29 * <br/>
30 * The method {@link #mapRow(ResultSet)} will be called for each row in the {@link ResultSet}, and it should
31 * return an instance of the object type {@code T} representing the mapped data from the row.
32 *
33 * @param <T> The type of object that each row should be mapped to.
34 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
35 * @since 1.0
36 * @version 1.0
37 * @see web.ambassador.db.QueryBuilder
38 * @see web.ambassador.db.DatabaseManager
39 */
40public interface RowMapper<T> {
41 /**
42 * Maps a single row of the {@code ResultSet} to an object of type {@code T}.
43 *
44 * This method is invoked for each row in the {@code ResultSet}. It is responsible for extracting the necessary
45 * data from the {@code ResultSet} and converting it into an object of type {@code T}.
46 *
47 * @param rs The {@code ResultSet} containing the current row of data to be mapped.
48 * @return The object of type {@code T} representing the mapped row data.
49 * @throws SQLException If an error occurs while accessing the data in the {@code ResultSet}.
50 */
51 T mapRow(ResultSet rs) throws SQLException;
52}
