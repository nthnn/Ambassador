24/**
25 * The {@code DefaultErrorController} is a default implementation of the {@code ErrorController} interface.
26 * It is responsible for handling HTTP errors and providing appropriate error messages and responses.
27 * This implementation maps standard HTTP status codes to their corresponding messages and
28 * returns an HTML representation of the error using a DOM structure.
29 * <br/>
30 * When an error occurs in the web application, this controller generates a response that
31 * includes the HTTP status code and a message that describes the error. The response is
32 * typically displayed to the user in the browser or logged for debugging purposes.
33 * <br/>
34 * The controller maps error codes (such as 404 for "Not Found" or 500 for "Internal Server Error")
35 * to their respective messages. If a status code is unrecognized, it returns a generic error message.
36 *
37 * <p><b>Usage Example:</b></p>
38 * ```java
39 * DefaultErrorController errorController = new DefaultErrorController();
40 * Dom errorPage = errorController.handleError(404, request, response);
41 * // Render or return the generated error page
42 * ```
43 *
44 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
45 * @version 1.0
46 * @since 1.0
47 */
48public class DefaultErrorController implements ErrorController {
49 /**
50 * Handles HTTP errors by generating an HTML error page with the status code and
51 * a corresponding error message.
52 *
53 * This method takes the HTTP status code, request, and response objects,
54 * maps the status code to a standard message, and returns a DOM representation
55 * of the error page.
56 *
57 * The method uses a switch statement to map the provided status code to a
58 * corresponding message. If the status code is not explicitly listed, it defaults
59 * to a generic "Error" message followed by the status code.
60 *
61 * @param statusCode The HTTP status code representing the error (e.g., 404 for "Not Found").
62 * @param request The HTTP request that caused the error. This can be used to inspect the request context.
63 * @param response The HTTP response that will be sent to the client. This can be used to set response headers.
64 * @return A {@code Dom} object representing the error page, which includes the status code and message.
65 */
66 @Override
67 public Dom handleError(int statusCode, Request request, Response response) {
68 String message = switch(statusCode) {
69 case 100 -> "Continue";
70 case 101 -> "Switching Protocols";
71 case 102 -> "Processing";
72 case 103 -> "Early Hints";
73 case 201 -> "Created";
74 case 202 -> "Accepted";
75 case 203 -> "Non-Authoritative Information";
76 case 204 -> "No Content";
77 case 205 -> "Reset Content";
78 case 206 -> "Partial Content";
79 case 207 -> "Multi-Status";
80 case 208 -> "Already Reported";
81 case 226 -> "IM Used";
82 case 300 -> "Multiple Choices";
83 case 301 -> "Moved Permanently";
84 case 302 -> "Found";
85 case 303 -> "See Other";
86 case 304 -> "Not Modified";
87 case 305 -> "Use Proxy";
88 case 306 -> "Unused";
89 case 307 -> "Temporary Redirect";
90 case 308 -> "Permanent Redirect";
91 case 400 -> "Bad Request";
92 case 401 -> "Unauthorized";
93 case 402 -> "Payment Required";
94 case 403 -> "Forbidden";
95 case 404 -> "Page Not Found";
96 case 405 -> "Method Not Allowed";
97 case 406 -> "Not Acceptable";
98 case 407 -> "Proxy Authentication Required";
99 case 408 -> "Request Timeout";
100 case 409 -> "Conflict";
101 case 410 -> "Gone";
102 case 411 -> "Length Required";
103 case 412 -> "Precondition Failed";
104 case 413 -> "Content Too Large";
105 case 414 -> "URI Too Long";
106 case 415 -> "Unsupported Media Type";
107 case 416 -> "Range Not Satisfiable";
108 case 417 -> "Expectation Failed";
109 case 418 -> "I'm A Teapot";
110 case 421 -> "Misdirected Request";
111 case 422 -> "Unprocessable Content";
112 case 423 -> "Locked";
113 case 424 -> "Failed Dependency";
114 case 425 -> "Too Early";
115 case 426 -> "Upgrade Required";
116 case 428 -> "Precondition Required";
117 case 429 -> "Too Many Requests";
118 case 431 -> "Request Header Fields Too Large";
119 case 451 -> "Unavailable For Legal Reasons";
120 case 500 -> "Internal Server Error";
121 case 501 -> "Not Implemented";
122 case 502 -> "Bad Gateway";
123 case 503 -> "Service Unavailable";
124 case 504 -> "Gateway Timeout";
125 case 505 -> "HTTP Version Not Supported";
126 case 506 -> "Variant Also Negotiates";
127 case 507 -> "Insufficient Storage";
128 case 508 -> "Loop Detected";
129 case 510 -> "Not Extended";
130 case 511 -> "Network Authentication Required";
131 default -> "Error " + statusCode;
132 };
133
134 return Dom.div(
135 Dom.h1(String.valueOf(statusCode)),
136 Dom.p(message)
137 );
138 }
