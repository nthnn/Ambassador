27/**
28 * Annotation to define the HTTP method type for a handler method in the Ambassador framework.
29 *
30 * This annotation is used to specify the HTTP method (GET, POST, PUT, DELETE, etc.)
31 * that a particular method in a controller should respond to. By default, if no value
32 * is provided, the method will handle HTTP GET requests.
33 *
34 * <p><b>Example Usage:</b></p>
35 * ```java
36 * @Controller(path = "/users")
37 * public class UserController implements Component{
38 * @Method(MethodType.GET)
39 * public ViewContent index(Request request, Response response, DatabaseManager dbManager) {
40 * // Logic to fetch user list
41 * }
42 *
43 * @Method(MethodType.POST)
44 * public ViewContent createUser(Request request, Response response, DatabaseManager dbManager) {
45 * // Logic to create a new user
46 * }
47 * }
48 * ```
49 *
50 * This annotation works in conjunction with the {@code Controller} annotation to define
51 * RESTful endpoints. It helps map methods to specific HTTP methods, allowing the framework
52 * to route requests appropriately.
53 *
54 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
55 * @version 1.0
56 * @since 1.0
57 * @see Controller
58 * @see web.ambassador.enums.MethodType
59 */
60@Retention(RetentionPolicy.RUNTIME)
61@Target(ElementType.METHOD)
62public @interface Method {
63 /**
64 * The HTTP method type that the annotated method should handle.
65 * By default, the method will handle GET requests.
66 *
67 * @return the HTTP method type (e.g., GET, POST, PUT)
68 */
69 MethodType value() default MethodType.GET;
70}
