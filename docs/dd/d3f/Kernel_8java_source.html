30/**
31 * Primary orchestrator for HTTP server management, routing, and database connection pooling.
32 * The Kernel class encapsulates the core logic of starting and managing an HTTP server using Java's
33 * built-in {@code HttpServer} class. It handles request routing, manages thread pools for request execution,
34 * and facilitates interaction with the database layer via the {@code DatabaseManager}.
35 *
36 * <p><b>Key features include:</b></p>
37 * <ul>
38 * <li>Configurable thread pool management</li>
39 * <li>Dynamic controller registration for routing</li>
40 * <li>Graceful shutdown with automatic resource cleanup</li>
41 * <li>Customizable error handling for HTTP responses</li>
42 * <li>Database connection pooling and management</li>
43 * </ul>
44 *
45 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
46 * @version 1.0
47 * @since 1.0
48 */

165 /**
166 * Registers shutdown hooks to ensure clean server termination.
167 *
168 * @param onSuccess Runnable to execute upon successful shutdown.
169 * @param onFail Runnable to execute if shutdown encounters an error.
170 */
171 public void setupShutdownHook(Runnable onSuccess, Runnable onFail) {
172 Runtime.getRuntime().addShutdownHook(new Thread(()-> {
173 this.cleanableSuccessRunner = this.cleaner.register(this, onSuccess);
174 this.cleanableFailRunner = this.cleaner.register(this, onFail);
175
176 try {
177 Kernel.this.stop();
178 onSuccess.run();
179 }
180 catch(Exception e) {
181 onFail.run();
182 }
183 }));
184 }

223 /**
224 * Starts the HTTP server and initializes the request processing pipeline.
225 *
226 * This method sets up the HTTP server to listen on the specified port, creates a thread pool
227 * for handling incoming requests, and registers the root request handler (`KernelHandler`)
228 * to manage routing and error handling.
229 *
230 * The server operates asynchronously, processing multiple requests concurrently based on
231 * the configured thread pool size.
232 *
233 * @throws IOException if the server fails to start or encounters binding issues.
234 *
235 * <p><b>Example:</b></p>
236 * ```java
237 * kernel.start();
238 * ```
239 */
240 public void start() throws IOException {
241 HttpServer server = HttpServer.create(
242 new InetSocketAddress(this.port),
243 this.backlog
244 );
245
246 this.executor = new ThreadPoolExecutor(
247 this.maxThreads / 2,
248 this.maxThreads,
249 60L,
250 TimeUnit.SECONDS,
251 new ArrayBlockingQueue<>(100),
252 new ThreadPoolExecutor.CallerRunsPolicy()
253 );
254
255 server.createContext("/", KernelHandler.build(
256 this.router,
257 this.dbManager,
258 this.errorControllers
259 ));
260 server.setExecutor(this.executor);
261 server.start();
262
263 System.out.println(
264 "Server started on port " + this.port +
265 " with " + this.maxThreads + " threads"
266 );
267 }

269 /**
270 * Stops the HTTP server and shuts down associated resources gracefully.
271 *
272 * This method initiates the shutdown of the thread pool and closes all active database connections.
273 * It ensures that ongoing tasks are completed within a specified timeout period (60 seconds).
274 * If tasks do not terminate within this period, a forceful shutdown is triggered.
275 *
276 * This method is invoked automatically during shutdown or can be called explicitly.
277 *
278 * <p><b>Example:</b></p>
279 * ```java
280 * kernel.stop();
281 * ```
282 */
283 public void stop() {
284 if(this.executor == null)
285 return;
286 this.executor.shutdown();
287
288 if(this.dbManager != null)
289 this.dbManager.shutdown();
290
291 try {
292 if(!this.executor.awaitTermination(60, TimeUnit.SECONDS) &&
293 !this.executor.isShutdown())
294 this.executor.shutdownNow();
295 }
296 catch(InterruptedException e) {
297 this.executor.shutdownNow();
298 }
299 }
300
301 /**
302 * Closes the Kernel and performs resource cleanup.
303 *
304 * This method is part of the AutoCloseable interface, ensuring that resources are cleaned
305 * up properly when the Kernel instance is closed. It triggers any registered success or
306 * failure cleanup actions through the Cleaner API.
307 *
308 * <p><b>Example:</b></p>
309 * ```java
310 * try (Kernel kernel = new Kernel(8080)) {
311 * kernel.start();
312 * } // Kernel will be automatically closed here
313 * ```
314 */
315 @Override
316 public void close() {
317 if(this.cleanableFailRunner != null)
318 this.cleanableFailRunner.clean();
319
320 if(this.cleanableSuccessRunner != null)
321 this.cleanableSuccessRunner.clean();
322 }
