20import java.io.FileInputStream;
21import java.io.IOException;
22import java.util.Properties;
23
24/**
25 * The {@code KernelConfig} class represents the configuration settings required to initialize and run the web server.
26 *
27 * This class is used to encapsulate various configuration properties such as the server port, database connection details,
28 * thread pool settings, and more. These properties are typically read from an external properties file, which can be customized
29 * to modify the behavior of the server without needing to modify the code directly.
30 *
31 * The class is implemented as a {@code record}, which provides a concise way of declaring a class with immutable fields.
32 * All fields in this class are final and cannot be modified after they are set during object construction.
33 *
34 * It supports loading these properties from a properties file, where each key corresponds to a configuration item (e.g., "port",
35 * "db_url", etc.). The {@code load} method provides the logic to read the properties file, parse the values, and return an instance
36 * of {@code KernelConfig}.
37 *
38 * <p><b>Usage Example:</b></p>
39 * ```java
40 * try {
41 * KernelConfig config = KernelConfig.load("config.properties");
42 * System.out.println("Server will run on port: " + config.port());
43 * } catch (IOException e) {
44 * e.printStackTrace();
45 * }
46 * ```
47 *
48 * @param port The port number that the server will listen on. Default is 80 if not provided in the properties file.
49 * @param backlog The maximum number of pending connections in the server's backlog queue. Default is 100 if not provided in the properties file.
50 * @param dbUrl The URL of the database the server will connect to. Default is an empty string if not provided in the properties file.
51 * @param dbUsername The username for authenticating with the database. Default is an empty string if not provided in the properties file.
52 * @param dbPassword The password for authenticating with the database. Default is an empty string if not provided in the properties file.
53 * @param dbTimeout The timeout duration (in seconds) for database operations. Default is 60 seconds if not provided in the properties file.
54 * @param maxThreads The maximum number of threads that the server will use for processing requests. Default is 8 if not provided in the properties file.
55 * @param poolSize The size of the thread pool for handling server requests. Default is 2 if not provided in the properties file.
56 *
57 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
58 * @version 1.0
59 * @since 1.0
60 */
61public record KernelConfig(
62 int port,
63 int backlog,
64 String dbUrl,
65 String dbUsername,
66 String dbPassword,
67 int dbTimeout,
68 int maxThreads,
69 int poolSize
70) {
71 /**
72 * Loads the server configuration from a specified properties file.
73 * <br/>
74 * This method reads the properties file, parses its contents, and creates a {@code KernelConfig} instance.
75 * It uses default values for any properties that are not present in the file.
76 *
77 * @param propertiesFile The path to the properties file containing the server configuration.
78 * @return A {@code KernelConfig} instance containing the loaded configuration values.
79 * @throws IOException If there is an error reading the properties file.
80 *
81 * @see Properties
82 * @see KernelConfig
83 */
84 public static KernelConfig load(String propertiesFile) throws IOException {
85 Properties config = new Properties();
86 config.load(new FileInputStream(propertiesFile));
87
88 return new KernelConfig(
89 Integer.parseInt(config.getOrDefault("port", "80").toString()),
90 Integer.parseInt(config.getOrDefault("backlog", "100").toString()),
91 config.getOrDefault("db_url", "").toString(),
92 config.getOrDefault("db_username", "").toString(),
93 config.getOrDefault("db_password", "").toString(),
94 Integer.parseInt(config.getOrDefault("db_timeout", "60").toString()),
95 Integer.parseInt(config.getOrDefault("max_threads", "8").toString()),
96 Integer.parseInt(config.getOrDefault("pool_size", "2").toString())
97 );
98 }
99}
