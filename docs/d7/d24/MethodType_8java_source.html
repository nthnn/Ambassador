20/**
21 * Enum representing the various HTTP methods supported by the Ambassador framework.
22 * <p>
23 * This enum defines the standard HTTP methods used to indicate the desired action
24 * to be performed for a given resource. It is commonly used in request handling
25 * to specify the type of operation being performed.
26 * </p>
27 *
28 * <p><b>Supported Methods:</b></p>
29 * <ul>
30 * <li>{@code GET} - Retrieve information from the server (read-only)</li>
31 * <li>{@code POST} - Submit data to be processed (create or update)</li>
32 * <li>{@code PUT} - Update an existing resource</li>
33 * <li>{@code DELETE} - Remove a resource</li>
34 * <li>{@code PATCH} - Partially update a resource</li>
35 * <li>{@code CONNECT} - Establish a tunnel to the server</li>
36 * <li>{@code HEAD} - Retrieve metadata of a resource (headers only)</li>
37 * <li>{@code OPTIONS} - Describe communication options for the target resource</li>
38 * <li>{@code TRACE} - Perform a message loop-back test along the path to the resource</li>
39 * </ul>
40 *
41 * This enum is useful for routing, request processing, and building RESTful APIs,
42 * ensuring that the correct HTTP method is used for the desired operation.
43 *
44 * <p><b>Example Usage:</b></p>
45 * ```java
46 * @Method(MethodType.GET)
47 * public ViewContent index(Request request, Response response, DatabaseManager dbManager) {
48 * // Your index's code here...
49 * return null;
50 * }
51 * ```
52 *
53 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
54 * @version 1.0
55 * @since 1.0
56 */
57public enum MethodType {
58 /**
59 * Requests a representation of the specified resource.
60 * This method is used to retrieve data from the server.
61 */
62 GET,
63
64 /**
65 * Submits data to the specified resource.
66 * Often used to create a new resource or to submit data that requires processing.
67 */
68 POST,
69
70 /**
71 * Replaces all current representations of the specified resource with the provided data.
72 * Used to update or create a resource at the specified URL.
73 */
74 PUT,
75
76 /**
77 * Deletes the specified resource.
78 * This method is used to remove a resource from the server.
79 */
80 DELETE,
81
82 /**
83 * Partially modifies the resource.
84 * Used to update or modify a resource with a subset of data.
85 */
86 PATCH,
87
88 /**
89 * Establishes a tunnel to the server identified by the target resource.
90 * Typically used for establishing a connection to a web server using a proxy.
91 */
92 CONNECT,
93
94 /**
95 * Requests the headers of the specified resource, but without the body content.
96 * Often used to gather metadata about a resource, such as the length of the content.
97 */
98 HEAD,
99
100 /**
101 * Describes the communication options for the target resource.
102 * Used to query the available methods and capabilities supported by the server for a specific resource.
103 */
104 OPTIONS,
105
106 /**
107 * Performs a diagnostic trace of the path a request takes to the target resource.
108 * This method is useful for debugging and tracing the request-response cycle.
109 */
110 TRACE
111}
