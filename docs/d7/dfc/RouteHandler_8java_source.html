21import web.ambassador.http.Request;
22import web.ambassador.http.Response;
23import web.ambassador.view.Dom;
24import web.ambassador.view.ViewContent;

26import java.lang.reflect.Method;
27
28/**
29 * The {@code RouteHandler} class is responsible for invoking controller methods
30 * that handle HTTP requests, providing a bridge between incoming requests and
31 * the corresponding business logic in a web application.
32 *
33 * It leverages reflection to instantiate controller classes and call handler methods
34 * dynamically. This design allows for flexible and decoupled routing and controller
35 * handling.
36 *
37 * <p><b>Key Responsibilities:</b></p>
38 * <ul>
39 * <li>Instantiate the appropriate controller.</li>
40 * <li>Invoke the handler method with request, response, and database context.</li>
41 * <li>Return a {@code ViewContent} object representing the rendered view.</li>
42 * <li>Handle errors gracefully by returning a 500 Internal Server Error response.</li>
43 * </ul>
44 *
45 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
46 * @version 1.0
47 * @since 1.0
48 */
49public class RouteHandler {
50 private final Class<?> controllerClass;
51 private final Method handlerMethod;
52
53 /**
54 * Constructs a {@code RouteHandler} with a target controller and handler method.
55 *
56 * @param controllerClass The class of the controller that contains the handler method.
57 * @param handlerMethod The specific method to handle the request.
58 */
59 public RouteHandler(Class<?> controllerClass, Method handlerMethod) {
60 this.controllerClass = controllerClass;
61 this.handlerMethod = handlerMethod;
62 }
63
64 /**
65 * Handles an incoming HTTP request by invoking the associated controller method.
66 *
67 * This method performs the following steps:
68 * <ul>
69 * <li>Instantiates the controller using reflection.</li>
70 * <li>Invokes the handler method with {@code Request}, {@code Response},
71 * and {@code DatabaseManager} as arguments.</li>
72 * <li>Returns the resulting {@code ViewContent} object representing the
73 * rendered HTML view or API response.</li>
74 * </ul>
75 *
76 * If an exception occurs during method invocation, it sets the response status
77 * to 500 and returns an error page with the exception message.
78 *
79 * @param request The HTTP request object.
80 * @param response The HTTP response object.
81 * @param dbManager The database manager providing data access.
82 * @return A {@code ViewContent} representing the result of the request
83 * (either the actual view or an error page).
84 */
85 public ViewContent handle(Request request, Response response, DatabaseManager dbManager) {
86 try {
87 Object controller = controllerClass.getDeclaredConstructor()
88 .newInstance();
89
90 return (ViewContent) handlerMethod.invoke(
91 controller,
92 request,
93 response,
94 dbManager
95 );
96 }
97 catch(Exception e) {
98 response.setStatusCode(500);
99 return Dom.div(
100 Dom.h1("500 Internal Server Error"),
101 Dom.p("<p>" + e.getMessage() + "</p>")
102 );
103 }
104 }
105}
