34/**
35 * Manages database connections, providing connection pooling and transaction handling.
36 *
37 * The {@code DatabaseManager} class maintains a pool of database connections and provides methods
38 * for obtaining, releasing, and executing queries with transactions. It manages a connection pool
39 * that helps improve performance by reusing database connections.
40 * <br/>
41 * This class supports executing database queries with automatic transaction management, as well
42 * as a shutdown process that cleans up resources.
43 *
44 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
45 * @since 1.0
46 * @version 1.0
47 * @see web.ambassador.db.QueryBuilder
48 */
49public class DatabaseManager {
50 private final String url;
51 private final String username;
52 private final String password;
53 private final int maxPoolSize;
54 private final int timeoutSeconds;
55 private final BlockingQueue<Connection> connectionPool;
56 private final Set<Connection> usedConnections;
57 private volatile boolean isShuttingDown;
58
59 /**
60 * Constructs a {@code DatabaseManager} with the specified database connection details and default pool size and timeout.
61 *
62 * @param url The URL of the database.
63 * @param username The username for the database connection.
64 * @param password The password for the database connection.
65 */
66 public DatabaseManager(String url, String username, String password) {
67 this(url, username, password, 10, 30);
68 }
69
70 /**
71 * Constructs a {@code DatabaseManager} with the specified database connection details, pool size, and timeout.
72 *
73 * @param url The URL of the database.
74 * @param username The username for the database connection.
75 * @param password The password for the database connection.
76 * @param poolSize The maximum size of the connection pool.
77 * @param timeoutSeconds The maximum time (in seconds) to wait for a connection from the pool.
78 */
79 public DatabaseManager(String url, String username, String password, int poolSize, int timeoutSeconds) {
80 this.url = url;
81 this.username = username;
82 this.password = password;
83 this.maxPoolSize = poolSize;
84 this.timeoutSeconds = timeoutSeconds;
85 this.connectionPool = new ArrayBlockingQueue<>(poolSize);
86 this.usedConnections = Collections.synchronizedSet(new HashSet<>());
87 this.isShuttingDown = false;
88
89 this.initializePool();
90 }
91
92 /**
93 * Initializes the connection pool by creating the initial set of database connections.
94 *
95 * This method is invoked during the construction of the {@code DatabaseManager} and creates the
96 * specified number of connections that will be used by the pool.
97 *
98 * @throws DatabaseException If there is an error establishing connections to the database.
99 */
100 private void initializePool() {
101 try {
102 for(int i = 0; i < maxPoolSize; i++)
103 this.connectionPool.offer(createConnection());
104 }
105 catch(SQLException e) {
106 throw new DatabaseException("Failed to initialize connection pool", e);
107 }
108 }
109
110 /**
111 * Creates a new database connection using the specified connection details.
112 *
113 * @return A new {@code Connection} to the database.
114 * @throws SQLException If an error occurs while creating the connection.
115 */
116 private Connection createConnection() throws SQLException {
117 Connection conn = DriverManager.getConnection(
118 this.url,
119 this.username,
120 this.password
121 );
122
123 conn.setAutoCommit(true);
124 return conn;
125 }
126
127 /**
128 * Retrieves a {@code Connection} from the connection pool. Blocks until a connection is available or the timeout is reached.
129 *
130 * @return A {@code Connection} from the pool.
131 * @throws DatabaseException If there is an error retrieving the connection from the pool or if the timeout is reached.
132 */
133 public Connection getConnection() throws DatabaseException {
134 try {
135 Connection connection = this.connectionPool.poll(
136 this.timeoutSeconds,
137 TimeUnit.SECONDS
138 );
139
140 if(connection == null)
141 throw new DatabaseException("Timeout waiting for database connection");
142
143 if(!connection.isValid(1))
144 connection = createConnection();
145
146 this.usedConnections.add(connection);
147 return connection;
148 }
149 catch(SQLException | InterruptedException e) {
150 throw new DatabaseException("Failed to get database connection", e);
151 }
152 }
153
154 /**
155 * Releases a {@code Connection} back into the pool if the system is not shutting down.
156 * If the system is shutting down, the connection is closed immediately.
157 *
158 * @param connection The connection to be released.
159 */
160 public void releaseConnection(Connection connection) {
161 if(connection != null && this.usedConnections.remove(connection)) {
162 if(!this.isShuttingDown)
163 this.connectionPool.offer(connection);
164 else closeConnection(connection);
165 }
166 }
167
168 /**
169 * Closes the given {@code Connection}.
170 *
171 * @param connection The connection to be closed.
172 */
173 private void closeConnection(Connection connection) {
174 try {
175 connection.close();
176 }
177 catch(SQLException _) { }
178 }
179
180 /**
181 * Shuts down the {@code DatabaseManager}, closing all connections and clearing the connection pool.
182 */
183 public void shutdown() {
184 this.isShuttingDown = true;
185
186 for(Connection conn : this.usedConnections)
187 this.closeConnection(conn);
188 this.usedConnections.clear();
189
190 Connection conn;
191 while((conn = this.connectionPool.poll()) != null)
192 this.closeConnection(conn);
193 }

195 /**
196 * Executes a database operation within a transaction.
197 *
198 * The method ensures that the database operation is executed within a transaction, which will be committed
199 * upon success, or rolled back in case of an exception.
200 *
201 * @param callback The callback that performs the database operation.
202 * @param <T> The return type of the callback.
203 * @return The result of the database operation.
204 * @throws DatabaseException If there is an error executing the transaction or rolling back the transaction.
205 */
206 public <T> T withTransaction(TransactionCallback<T> callback) throws DatabaseException {
207 Connection connection = null;
208
209 try {
210 connection = getConnection();
211 connection.setAutoCommit(false);
212
213 T result = callback.execute(connection);
214 connection.commit();
215
216 return result;
217 }
218 catch(SQLException e) {
219 try {
220 connection.rollback();
221 }
222 catch(SQLException rollbackEx) {
223 throw new DatabaseException("Failed to rollback transaction", rollbackEx);
224 }
225
226 throw new DatabaseException("Transaction failed", e);
227 }
228 finally {
229 if(connection != null) {
230 try {
231 connection.setAutoCommit(true);
232 }
233 catch (SQLException _) {}
234
235 this.releaseConnection(connection);
236 }
237 }
238 }
239
240 /**
241 * Executes a query using the specified {@code QueryBuilder} and maps the results to a list of objects using the specified {@code RowMapper}.
242 *
243 * @param queryBuilder The {@code QueryBuilder} that constructs the query to be executed.
244 * @param mapper The {@code RowMapper} that maps each row of the result set to an object.
245 * @param <T> The type of the objects in the result list.
246 * @return A list of objects mapped from the result set.
247 * @throws DatabaseException If there is an error executing the query or mapping the results.
248 */
249 public <T> List<T> query(QueryBuilder queryBuilder, RowMapper<T> mapper) throws DatabaseException {
250 return this.withTransaction(connection -> {
251 List<T> results = new ArrayList<>();
252 try(PreparedStatement stmt = queryBuilder.prepare(connection);
253 ResultSet rs = stmt.executeQuery()) {
254 while(rs.next())
255 results.add(mapper.mapRow(rs));
256 }
257
258 return results;
259 });
260 }
261
262 /**
263 * Executes a query that updates the database (e.g., INSERT, UPDATE, DELETE) using the specified {@code QueryBuilder}.
264 *
265 * @param queryBuilder The {@code QueryBuilder} that constructs the update query.
266 * @return The number of rows affected by the query.
267 * @throws DatabaseException If there is an error executing the update query.
268 */
269 public int execute(QueryBuilder queryBuilder) throws DatabaseException {
270 return this.withTransaction(connection -> {
271 try(PreparedStatement stmt = queryBuilder.prepare(connection)) {
272 return stmt.executeUpdate();
273 }
274 });
275 }
