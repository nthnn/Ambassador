27/**
28 * Represents an HTML DOM (Document Object Model) element that can be dynamically constructed
29 * and rendered as part of an HTTP response.
30 *
31 * The {@code Dom} class implements the {@code ViewContent} interface, allowing HTML
32 * elements to be programmatically generated, modified, and serialized into an HTML string.
33 * This class provides a flexible and fluent API to build complex HTML structures without
34 * manually writing raw HTML tags.
35 *
36 * <p>Common use cases for this class include:</p>
37 * <ul>
38 * <li>Generating HTML tables, forms, and component-based UIs.</li>
39 * <li>Creating dynamic server-rendered HTML pages.</li>
40 * <li>Embedding HTML fragments in responses for web applications.</li>
41 * </ul>
42 *
43 * The {@code Dom} class supports the addition of children, attributes, CSS classes,
44 * inline styles, and custom HTML content. Cached rendering optimizes performance by storing
45 * the serialized byte array of the generated HTML.
46 *
47 * @see web.ambassador.view.ViewContent
48 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
49 * @since 1.0
50 * @version 1.0
51 */

639 /**
640 * Serializes the DOM element and its children to an HTML string.
641 *
642 * @return A string representing the HTML structure of this element.
643 */
644 @Override
645 public String toString() {
646 StringBuilder html = new StringBuilder();
647
648 if(this.tagName != null) {
649 html.append("<")
650 .append(this.tagName);
651
652 if(!this.classes.isEmpty())
653 html.append(" class=\"")
654 .append(String.join(" ", this.classes))
655 .append("\"");
656
657 if(!this.styles.isEmpty()) {
658 html.append(" style=\"");
659 this.styles.forEach((k, v) ->
660 html.append(k)
661 .append(":")
662 .append(v)
663 .append(";")
664 );
665 html.append("\"");
666 }
667
668 this.attributes.forEach((k, v) -> {
669 if(v != null)
670 html.append(" ")
671 .append(k)
672 .append("=\"")
673 .append(v)
674 .append("\"");
675 else html.append(" ")
676 .append(k);
677 });
678
679 if(this.tagName.equals("img") ||
680 this.tagName.equals("link") ||
681 this.tagName.equals("br") ||
682 this.tagName.equals("base"))
683 html.append(" />");
684 else html.append(">");
685 }
686
687 if(this.textContent != null)
688 html.append(this.textContent);
689
690 for(ViewContent child : this.children)
691 html.append(child);
692
693 if(this.tagName != null &&
694 !this.tagName.equals("!DOCTYPE") &&
695 !this.tagName.equals("img") &&
696 !this.tagName.equals("link") &&
697 !this.tagName.equals("br") &&
698 !this.tagName.equals("base"))
699 html.append("</")
700 .append(this.tagName)
701 .append(">");
702
703 return html.toString();
704 }
