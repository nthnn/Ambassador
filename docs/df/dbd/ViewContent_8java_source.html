18package web.ambassador.view;

20/**
21 * Represents content that can be rendered as part of a web response.
22 *
23 * The {@code ViewContent} interface defines the contract for all types
24 * of content that can be returned as part of an HTTP response. Implementing
25 * classes are responsible for providing the content to be sent back to the
26 * client, as well as the MIME type of the content to correctly identify the
27 * format of the response.
28 *
29 * Typically, {@code ViewContent} implementations might include HTML pages,
30 * JSON responses, or other types of media such as images or text files.
31 * The interface provides methods to render the content in its raw form and
32 * to specify the MIME type to ensure that the response is handled correctly
33 * by the client (e.g., browser, mobile device, etc.).
34 *
35 * @see web.ambassador.view.Component
36 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
37 * @since 1.0
38 * @version 1.0
39 */
40public interface ViewContent {
41 /**
42 * Renders the content to be returned in the HTTP response.
43 *
44 * This method is responsible for generating the actual raw content of
45 * the response, which could be in various formats (e.g., HTML, JSON, etc.).
46 * The content is returned as a byte array, which allows for flexibility
47 * in how the content is represented (e.g., binary data, encoded text, etc.).
48 *
49 * @return A byte array representing the rendered content, ready to be sent
50 * as the HTTP response body.
51 */
52 byte[] render();
53
54 /**
55 * Returns the MIME type of the content.
56 *
57 * The MIME type specifies the format of the response content and helps
58 * the client interpret the data correctly. For example, {@code "text/html"}
59 * indicates that the response content is an HTML page, while {@code "application/json"}
60 * indicates that the content is in JSON format.
61 *
62 * @return A string representing the MIME type of the content (e.g., "text/html", "application/json").
63 */
64 String mimeType();
65}
