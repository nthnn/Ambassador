18package web.ambassador.core;
19
20/**
21 * The AssetResponse class represents the response returned by the AssetHandler when
22 * retrieving an asset (such as an image, CSS, or JavaScript file) from the classpath.
23 *
24 * This class encapsulates the asset content as a byte array, its MIME type, and
25 * a flag indicating whether the asset was successfully found and loaded.
26 *
27 * <p><b>Usage:</b></p>
28 * ```java
29 * AssetResponse response = assetHandler.getAsset("/assets/image.png");
30 * if(response.wasFound()) {
31 * byte[] content = response.getContent();
32 * String mimeType = response.getMimeType();
33 * // Process content and MIME type
34 * }
35 * ```
36 *
37 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
38 * @version 1.0
39 * @since 1.0
40 */
41public class AssetResponse {
42 private final byte[] content;
43 private final String mimeType;
44 private final boolean found;
45
46 /**
47 * Constructs a new AssetResponse.
48 *
49 * @param content The content of the asset as a byte array.
50 * @param mimeType The MIME type of the asset (e.g., "image/png", "text/css").
51 * @param found A boolean flag indicating whether the asset was found and successfully loaded.
52 */
53 public AssetResponse(byte[] content, String mimeType, boolean found) {
54 this.content = content;
55 this.mimeType = mimeType;
56 this.found = found;
57 }
58
59 /**
60 * Returns the content of the asset as a byte array.
61 *
62 * @return A byte array representing the content of the asset. Can be empty or null
63 * if the asset was not found.
64 */
65 public byte[] getContent() {
66 return this.content;
67 }
68
69 /**
70 * Returns the MIME type of the asset.
71 *
72 * @return A string representing the MIME type of the asset (e.g., "image/png").
73 * If the asset was not found, this will be null.
74 */
75 public String getMimeType() {
76 return this.mimeType;
77 }
78
79 /**
80 * Returns whether the asset was successfully found and loaded.
81 *
82 * @return A boolean flag indicating if the asset was found and successfully loaded.
83 * If true, the content is valid; if false, the content will be null.
84 */
85 public boolean wasFound() {
86 return this.found;
87 }
88}
