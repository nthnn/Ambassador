32/**
33 * The {@code KernelHandler} class is responsible for handling incoming HTTP requests and returning appropriate responses.
34 *
35 * This class implements the {@code HttpHandler} interface and is part of the server's request handling pipeline. It interacts with the {@code Router},
36 * which is used to process the request and determine the correct view/content to return. Additionally, it manages the database connection
37 * through the {@code DatabaseManager} and handles error responses using the provided {@code ErrorController} implementations.
38 *
39 * The {@code KernelHandler} class is designed to handle various HTTP methods (e.g., GET, POST), parse request headers, cookies, and query parameters,
40 * and generate corresponding responses. If an error occurs during request processing, it will handle the error gracefully by using the appropriate
41 * error controller.
42 *
43 * In the event of an exception or error, a default error handler is used if no specific error controller is provided for the given HTTP status code.
44 *
45 * @author <a href="https://github.com/nthnn" target="_blank">Nathanne Isip</a> &lt;<a href="mailto:nathanneisip@gmail.com">nathanneisip@gmail.com</a>&gt;
46 * @version 1.0
47 * @since 1.0
48 */
49public class KernelHandler implements HttpHandler {
50 private Router router = null;
51 private DatabaseManager dbManager = null;
52 private Map<Integer, ErrorController> errorControllers = null;
53
54 /**
55 * Private constructor for creating a {@code KernelHandler} instance.
56 */
57 protected KernelHandler() { }

59 /**
60 * Static method to build a new {@code KernelHandler} instance.

63 * {@code DatabaseManager}, and error controllers.

65 * @param router The router used for handling requests and determining view/content.

67 * @param errorControllers A map of error codes to their respective error controllers.
68 * @return A {@code KernelHandler} instance.

75 KernelHandler handler = new KernelHandler();
76 handler.router = router;

78 handler.errorControllers = errorControllers;

83 /**
84 * Handles an incoming HTTP request by parsing the request details, processing the request using the router,
85 * and returning an appropriate response.
86 *
87 * This method performs the following steps:
88 * <ol>
89 * <li>Creates a new {@code Request} and {@code Response} object.</li>
90 * <li>Parses request details such as path, method, headers, cookies, query parameters, and body (for POST requests).</li>
91 * <li>Processes the request using the {@code Router} to determine the appropriate view or content.</li>
92 * <li>If the response status code is not 200, an error response is generated using the appropriate error controller.</li>
93 * <li>Writes the response to the HTTP exchange, including headers and response body.</li>
94 * </ol>
95 *
96 * In case of an exception during request processing, the method will respond with a generic 500 Internal Server Error,
97 * using the error controller if defined.
98 *
99 * @param exchange The {@code HttpExchange} object representing the incoming HTTP request and response.
100 * @throws IOException If an I/O error occurs while handling the request or response.
101 */
102 @Override
103 public void handle(HttpExchange exchange) throws IOException {
104 Request request = new Request();
105 Response response = new Response();
106 InputStream requestBody = null;
107
108 try {
109 request.setPath(exchange.getRequestURI().getPath());
110 request.setMethod(MethodType.valueOf(exchange.getRequestMethod()));
111
112 exchange.getRequestHeaders().forEach((key, values) ->
113 request.getHeaders().put(key, values.getFirst())
114 );
115 request.parseCookies();
116
117 String query = exchange.getRequestURI().getQuery();
118 request.parseQueryString(query);
119
120 if(exchange.getRequestMethod().equalsIgnoreCase("POST")) {
121 requestBody = exchange.getRequestBody();
122 request.setBody(requestBody);
123 request.parseMultipartData();
124 }
125
126 byte[] responseBytes;
127 ViewContent result = this.router.handleRequest(
128 request,
129 response,
130 this.dbManager
131 );
132
133 if(response.getStatusCode() != 200) {
134 int statusCode = response.getStatusCode();
135 ErrorController controller = this.errorControllers
136 .getOrDefault(statusCode, new DefaultErrorController());
137 responseBytes = controller.handleError(statusCode, request, response)
138 .render();
139 }
140 else responseBytes = result.render();
141
142 request.getHeaders().entrySet().stream()
143 .filter(
144 entry ->
145 entry.getKey()
146 .equalsIgnoreCase("Set-Cookie")
147 )
148 .forEach(
149 cookie ->
150 response.setCookie(cookie.getKey(), cookie.getValue())
151 );
152
153 exchange.getResponseHeaders().add("Content-type", result.mimeType());
154 for(Map.Entry<String, String> header: response.getHeaders().entrySet())
155 exchange.getResponseHeaders().add(header.getKey(), header.getValue());
156
157 exchange.sendResponseHeaders(response.getStatusCode(), responseBytes.length);
158 exchange.getResponseBody().write(responseBytes);
159 }
160 catch(Exception e) {
161 int statusCode = response.getStatusCode();
162 ErrorController controller = this.errorControllers
163 .getOrDefault(statusCode, new DefaultErrorController());
164 byte[] errorBytes = controller.handleError(statusCode, request, response)
165 .render();
166
167 response.setStatusCode(500);
168 exchange.sendResponseHeaders(500, errorBytes.length);
169 exchange.getResponseBody().write(errorBytes);
170 }
171 finally {
172 if(requestBody != null)
173 try {
174 requestBody.close();
175 }
176 catch(IOException _) { }
177
178 exchange.getResponseBody().close();
179 }
180 }
